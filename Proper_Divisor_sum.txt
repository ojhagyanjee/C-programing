#include<stdio.h>
//#include<malloc.h>

void main()
{
    int n,a[100],x,i,j;
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
        scanf("%d",&a[i]);
    }
    for(i=1;i<=n;i++)
    {
        x=0;
        for(j=1;j<a[i];j++)
        {
            if(a[i]%j==0)
            {
                x+=j;
            }
        }
        if(x==a[i])
            {
                printf("YES\n");
            }
         else
            printf("NO\n");
    }
}
